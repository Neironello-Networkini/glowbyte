TABLE Supplies {
    id integer [PRIMARY KEY]
    brand_id INT // Код марки
    warehouse_id INT // Склад, идентификатор
    stack_number INT // Номер штабеля
    warehouse_date DATE  // Дата выгрузки на склад
    warehouse_weight DECIMAL(15, 4) // Вес на складе (тн)
    ship_date DATE // Дата погрузки на судно
    ship_weight DECIMAL(15, 4) // Вес на судне (тн)
}

Table Warehouse {
  id integer [PRIMARY KEY]
  name varchar(100)
}

Table Brand {
  id integer [PRIMARY KEY]
  name varchar(100)
}

TABLE Temperature {
    id integer [PRIMARY KEY]
    brand_id INT // Марка
    warehouse_id INT
    stack_number INT
    max_temperature DECIMAL(5, 2) // Максимальная температура
    picket VARCHAR(50) // Пикет
    act_date DATE // Дата акта
    shift INT // Смена
}


Ref: Supplies.warehouse_id > Warehouse.id
Ref: Temperature.warehouse_id > Warehouse.id
Ref: Supplies.brand_id > Brand.id
Ref: Temperature.brand_id > Brand.id
